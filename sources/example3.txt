import React, { useRef } from "react";
import { Canvas, useFrame, extend } from "@react-three/fiber";
import { shaderMaterial } from "@react-three/drei";
import * as THREE from "three";
import { motion } from "framer-motion";

// Define a custom shader material producing a subtle quantum glow and dynamic displacement.
const QuantumMaterial = shaderMaterial(
  // Uniforms
  {
    uTime: 0,
    uResolution: new THREE.Vector2(window.innerWidth, window.innerHeight)
  },
  // Vertex Shader
  `
    varying vec2 vUv;
    void main() {
      vUv = uv;
      vec3 pos = position;
      // Introduce a gentle, time-dependent wave for an organic 3D movement
      pos.z += sin(pos.x * 3.0 + uTime) * 0.1;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
    }
  `,
  // Fragment Shader
  `
    uniform float uTime;
    uniform vec2 uResolution;
    varying vec2 vUv;
    void main() {
      // Compute a centered radial gradient for a soft glow effect
      vec2 centered = vUv - 0.5;
      float dist = length(centered);
      float alpha = smoothstep(0.4, 0.0, dist);
      // Mix pure black with white glow at very low intensity for subtlety
      vec3 color = mix(vec3(0.0), vec3(1.0), alpha);
      gl_FragColor = vec4(color, alpha * 0.1);
    }
  `
);
extend({ QuantumMaterial });

// A 3D mesh component that uses the custom shader material and rotates slowly.
const QuantumMesh = () => {
  const meshRef = useRef<THREE.Mesh>(null!);
  const materialRef = useRef<any>(null);

  useFrame(({ clock }) => {
    const elapsedTime = clock.getElapsedTime();
    if (meshRef.current) {
      meshRef.current.rotation.y = elapsedTime * 0.1;
      meshRef.current.rotation.x = elapsedTime * 0.05;
    }
    if (materialRef.current) {
      materialRef.current.uTime = elapsedTime;
    }
  });

  return (
    <mesh ref={meshRef}>
      <icosahedronGeometry args={[1.5, 64]} />
      <quantumMaterial ref={materialRef} />
    </mesh>
  );
};

// Main Landing Page Component – a stunning scaffold merging a dynamic 3D background with minimal, elegant UI.
const LandingPage: React.FC = () => {
  return (

      </Canvas>
